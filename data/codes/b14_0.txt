<script src="https://cdn.jsdelivr.net/npm/swiper@11/swiper-bundle.min.js"></script>
<script>
    async function hostReactAppReady(selector = "#__next > div", timeout = 500, ) {
        return new Promise((resolve) => {
            const waiter = () => {
                const host_el = document.querySelector(selector);
                if (host_el?.getBoundingClientRect().height) {
                    resolve();
                } else {
                    setTimeout(waiter, timeout);
                }
            };
            waiter();
        });
    }
    hostReactAppReady().then(() => {
        document.querySelectorAll(".slider-block").forEach((block) => {
            const swiperEl = block.querySelector(".swiper");
            if (!swiperEl) return;
            const perView = parseInt(block.dataset.perView || "3", 10);
            const slidesCount = swiperEl.querySelectorAll(".swiper-slide").length;
            const swiper = new Swiper(swiperEl, {
                slidesPerView: 1,
                spaceBetween: 24,
                loop: slidesCount >= perView * 2,
                watchOverflow: true,
                breakpoints: {
                    768: {
                        slidesPerView: Math.min(2, perView),
                        spaceBetween: 24,
                    },
                    1024: {
                        slidesPerView: perView,
                        spaceBetween: 24,
                    },
                },
                pagination: {
                    el: block.querySelector(".swiper-pagination"),
                    clickable: true,
                },
                navigation: {
                    nextEl: block.querySelector(".slider__next"),
                    prevEl: block.querySelector(".slider__prev"),
                },
                on: {
                    lock() {
                        block.classList.add("slider-block--locked");
                    },
                    unlock() {
                        block.classList.remove("slider-block--locked");
                    },
                },
            });
            block.classList.toggle("slider-block--locked", swiper.isLocked);
        });
    });
</script>
